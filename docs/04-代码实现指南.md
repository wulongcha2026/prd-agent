## 3. 核心代码

### 3.1 状态定义 (state.py)

```python
from typing import TypedDict, Optional
from pydantic import BaseModel

class PRDState(TypedDict):
    """PRD Agent 状态"""
    raw_input: str                    # 原始输入
    intent: Optional[str]             # 需求类型: product/bug/optimize
    missing_info: list[str]            # 缺失信息
    requirements: dict                # 提取的需求要素
    prd_content: str                  # 生成的 PRD
    review_result: dict               # 审核结果
    iterations: int                   # 迭代次数

class Requirement(BaseModel):
    """需求要素"""
    user_persona: str        # 用户角色
    user_goal: str          # 用户目标
    user_value: str         # 用户价值
    scenarios: list[str]    # 使用场景
    acceptance: list[str]   # 验收标准
```

### 3.2 节点定义 (nodes.py)

```python
from langchain_core.messages import HumanMessage, AIMessage
from app.agent.state import PRDState
from app.core.llm import get_llm

def intent_recognition(state: PRDState) -> PRDState:
    """节点1：意图识别"""
    llm = get_llm()
    prompt = f"""分析以下需求，确定需求类型：
    - 产品需求：新功能开发
    - BUG 反馈：问题缺陷
    - 优化建议：性能/体验改进
    
    需求：{state['raw_input']}
    
    返回：product / bug / optimize"""
    
    intent = llm.invoke(prompt).content
    state['intent'] = intent
    return state

def information_gap(state: PRDState) -> PRDState:
    """节点2：信息补全"""
    llm = get_llm()
    prompt = f"""分析需求，识别缺失的关键信息：
    
    需求：{state['raw_input']}
    类型：{state['intent']}
    
    返回缺失信息列表（JSON数组），如：["用户角色", "目标用户"]
    如果信息完整，返回空数组：[]"""
    
    missing = llm.invoke(prompt).content
    state['missing_info'] = eval(missing) if missing != "[]" else []
    return state

def should_ask_more(state: PRDState) -> str:
    """判断是否需要追问"""
    if state['missing_info'] and state['iterations'] < 3:
        return "ask_more"
    return "continue"

def generate_prd(state: PRDState) -> PRDState:
    """节点3：生成 PRD"""
    llm = get_llm()
    prompt = f"""根据以下需求生成标准 PRD 文档：

需求：{state['raw_input']}
类型：{state['intent']}
要素：{state['requirements']}

输出格式要求：
```markdown
# [产品名称] 产品需求文档

## 1. 需求概述
## 2. 用户故事
## 3. 功能需求
## 4. 验收标准
```
"""
    prd = llm.invoke(prompt).content
    state['prd_content'] = prd
    return state
```

### 3.3 工作流组装 (workflow.py)

```python
from langgraph.graph import StateGraph, END
from app.agent.state import PRDState
from app.agent.nodes import (
    intent_recognition,
    information_gap,
    should_ask_more,
    generate_prd,
    ask_for_info
)

def create_workflow() -> StateGraph:
    """创建 Agent 工作流"""
    
    workflow = StateGraph(PRDState)
    
    # 添加节点
    workflow.add_node("intent", intent_recognition)
    workflow.add_node("gap_check", information_gap)
    workflow.add_node("ask_more", ask_for_info)
    workflow.add_node("generate", generate_prd)
    
    # 设置入口
    workflow.set_entry_point("intent")
    
    # 添加边
    workflow.add_edge("intent", "gap_check")
    
    # 条件边：判断是否需要追问
    workflow.add_conditional_edges(
        "gap_check",
        should_ask_more,
        {
            "ask_more": "ask_more",
            "continue": "generate"
        }
    )
    
    # 追问后回到信息检查
    workflow.add_edge("ask_more", "gap_check")
    
    # 生成后结束
    workflow.add_edge("generate", END)
    
    return workflow.compile()
```

### 3.4 API 接口 (routes.py)

```python
from fastapi import APIRouter
from pydantic import BaseModel

router = APIRouter(prefix="/api/v1/prd", tags=["PRD"])

class GenerateRequest(BaseModel):
    input: str
    template: str = "standard"

@router.post("/generate")
async def generate_prd(request: GenerateRequest):
    workflow = create_workflow()
    
    initial_state = PRDState(
        raw_input=request.input,
        missing_info=[],
        requirements={},
        prd_content="",
        review_result={},
        iterations=0
    )
    
    result = workflow.invoke(initial_state)
    
    return {
        "prd_id": "prd_xxx",
        "content": result['prd_content'],
        "intent": result['intent'],
        "missing_info": result.get('missing_info', [])
    }
```

## 4. Prompt 模板

### 4.1 意图识别 Prompt

```
你是一个需求分析师。请分析以下需求，确定需求类型。

需求：{input}

可选类型：
- product：新产品功能开发
- bug：问题缺陷反馈
- optimize：性能或体验优化

直接返回类型名称，不要其他内容。
```

### 4.2 PRD 生成 Prompt

```
你是一个资深产品经理。请根据以下需求信息生成标准的产品需求文档（PRD）。

需求类型：{intent}
需求描述：{raw_input}
提取的要素：{requirements}

请按以下格式输出：

# [产品名称] 产品需求文档

## 1. 需求概述
## 2. 用户故事
## 3. 功能需求
## 4. 非功能需求
## 5. 验收标准
## 6. 依赖关系

要求：
- 内容完整、专业
- 适合开发团队阅读
- 验收标准可测试
```
